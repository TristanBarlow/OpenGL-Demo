#version 330 core
out vec4 colour;

in vec4 vertexColourOut;
in vec2 vertexTextureCoordOut;
in vec3 vertexNormalDir;
in vec3 vertCameraDir;

uniform sampler2D baseTexture;

uniform vec3 lightDirection;
uniform vec4 diffuseLightColour;
uniform float lightDistance;
uniform vec4 diffuseMaterialColour;


vec3 shine;


void main()
{
	float shineness = dot(vertexNormalDir, vertCameraDir);
	shineness = pow(clamp(shineness, 0, 1), 50.0);
	vec4 specular = shineness* vec4(1.0f,1.0f,1.0f,1.0f) * vec4(1.0f,1.0f,1.0f,1.0f);

	float cosTheta = dot(vertexNormalDir, lightDirection );
	cosTheta = clamp(cosTheta, 0,1);
    vec4 diffuse = cosTheta*texture2D(baseTexture,vertexTextureCoordOut);

  //colour = fragmentColour;
  colour= diffuse+ specular;

}
