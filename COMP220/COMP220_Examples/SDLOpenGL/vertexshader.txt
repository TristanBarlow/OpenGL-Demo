#version 330 core
layout(location = 0) in vec3 vertexPos;
layout(location = 1) in vec3 vertexColor;

out vec3 vertexColorOut;

uniform vec2 vertexOffset = vec2(0.0);

void main()
{
vertexColorOut = vertexColor;
vec3 offsetVertexPos = vertexPos;
offsetVertexPos.xy += vertexOffset;
gl_Position = vec4 (offsetVertexPos, 1.0);

//gl_Position.xyz = vertexPos;
//gl_Position.w = 1.0;
}
